# Start a pod with 2.1 torch and 11.8 cuda 
# 1. Clone the repo
git clone https://github.com/google/STORM.git
cd STORM

# 2. Create venv with Python 3.10
python3.10 -m venv --system-site-packages venv
source venv/bin/activate

# 3. Confirm correct python
which python
python --version  # should say 3.10.x

# 4. Install torch first, matched to CUDA 11.8
# pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/cu118

# 5. Install numpy and opencv pinned to compatible versions
pip install "numpy>=1.26,<2.0" opencv-python==4.8.1.78

# 6. Install all other dependencies
pip install \
  albumentations \
  pudb \
  invisible-watermark \
  imageio \
  imageio-ffmpeg \
  pytorch-lightning \
  omegaconf \
  test-tube \
  streamlit \
  einops \
  torch-fidelity \
  torchmetrics \
  kornia \
  spacy \
  huggingface_hub==0.25.2 \
  transformers==4.26.0 \
  diffusers==0.12.1 \
  tokenizers==0.13.3 \
  ftfy \
  ipywidgets \
  matplotlib \
  pyrallis \
  accelerate \
  jupyter

# 7. Install git-based dependencies
pip install git+https://github.com/CompVis/taming-transformers.git@master
pip install git+https://github.com/openai/CLIP.git@main

# 8. Register venv as Jupyter kernel
pip install ipykernel
python -m ipykernel install --user --name=storm --display-name "storm"
